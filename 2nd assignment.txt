S = "ATTGTCCCAATCTGTTG"

dinucleotides = []
trinucleotides = []

for i in range(len(S) - 1):
    dinuc = S[i:i+2]
    if dinuc not in dinucleotides:
        dinucleotides.append(dinuc)

for i in range(len(S) - 2):
    trinuc = S[i:i+3]
    if trinuc not in trinucleotides:
        trinucleotides.append(trinuc)

print("Dinucleotides found:", dinucleotides)
print("Trinucleotides found:", trinucleotides)
